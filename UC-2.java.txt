import java.util.Random;

public class TicTacToe {

    static char player1Symbol;
    static char player2Symbol;
    static char currentPlayer;

    // UC: Toss + Symbol Assignment
    public static void initializeGame() {
        Random rand = new Random();

        int toss = rand.nextInt(2); // 0 or 1

        if (toss == 0) {
            // Player 1 starts
            player1Symbol = 'X';
            player2Symbol = 'O';
            currentPlayer = player1Symbol;

            System.out.println("Player 1 won the toss!");
            System.out.println("Player 1 is X and starts first.");
            System.out.println("Player 2 is O.");
        } else {
            // Player 2 starts
            player2Symbol = 'X';
            player1Symbol = 'O';
            currentPlayer = player2Symbol;

            System.out.println("Player 2 won the toss!");
            System.out.println("Player 2 is X and starts first.");
            System.out.println("Player 1 is O.");
        }
    }

    public static void main(String[] args) {
        initializeGame();

        System.out.println("Current Player Symbol: " + currentPlayer);
    }
}
